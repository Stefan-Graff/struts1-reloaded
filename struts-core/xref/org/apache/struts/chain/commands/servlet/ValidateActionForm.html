21  package org.apache.struts.chain.commands.servlet;
22  
23  import org.apache.struts.action.ActionErrors;
24  import org.apache.struts.action.ActionForm;
25  import org.apache.struts.action.ActionMapping;
26  import org.apache.struts.chain.commands.AbstractValidateActionForm;
27  import org.apache.struts.chain.contexts.ActionContext;
28  import org.apache.struts.chain.contexts.ServletActionContext;
29  import org.apache.struts.config.ActionConfig;
30  import org.slf4j.Logger;
31  import org.slf4j.LoggerFactory;
32  
33  /**
34   * <p>Validate the properties of the form bean for this request.  If there are
35   * any validation errors, execute the child commands in our chain; otherwise,
36   * proceed normally.  Also, if any errors are found and the request is a
37   * multipart request, rollback the <code>MultipartRequestHandler</code>.</p>
38   *
39   * @version $Rev$ $Date: 2005-06-04 10:58:46 -0400 (Sat, 04 Jun 2005)
40   *          $
41   */
42  public class ValidateActionForm extends AbstractValidateActionForm {
43      // ------------------------------------------------------ Instance Variables
44  
45      /**
46       * The {@code Log} instance for this class.
47       */
48      private final Logger log =
49          LoggerFactory.getLogger(ValidateActionForm.class);
50  
51      // ------------------------------------------------------- Protected Methods
52  
53      /**
54       * <p>Call the <code>validate()</code> method of the specified form bean,
55       * and return the resulting <code>ActionErrors</code> object.</p>
56       *
57       * @param context    The context for this request
58       * @param actionForm The form bean for this request
59       */
60      protected ActionErrors validate(ActionContext context,
61          ActionConfig actionConfig, ActionForm actionForm) {
62          ServletActionContext saContext = (ServletActionContext) context;
63          ActionErrors errors =
64              (actionForm.validate((ActionMapping) actionConfig,
65                  saContext.getRequest()));
66  
67          // Special handling for multipart request
68          if ((errors != null) && !errors.isEmpty()) {
69              if (actionForm.getMultipartRequestHandler() != null) {
70                  log.trace("  Rolling back multipart request");
71  
72                  actionForm.getMultipartRequestHandler().rollback();
73              }
74          }
75  
76          return errors;
77      }
78  }
