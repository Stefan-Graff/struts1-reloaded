21  package org.apache.struts.chain.commands;
22  
23  import org.apache.struts.action.Action;
24  import org.apache.struts.action.ActionMapping;
25  import org.apache.struts.chain.Constants;
26  import org.apache.struts.chain.commands.util.ClassUtils;
27  import org.apache.struts.chain.contexts.ActionContext;
28  import org.apache.struts.config.ActionConfig;
29  import org.apache.struts.config.ForwardConfig;
30  import org.apache.struts.dispatcher.Dispatcher;
31  
32  import java.util.HashMap;
33  import java.util.Map;
34  
35  import org.apache.commons.logging.Log;
36  import org.apache.commons.logging.LogFactory;
37  
38  public class ExecuteDispatcher extends ActionCommandBase {
39  
40      private static final Log log = LogFactory.getLog(ExecuteDispatcher.class);
41  
42      private String defaultDispatcherType;
43  
44      /**
45       * Creates the dispatcher of the specified type.
46       *
47       * @param type the dispatcher class name
48       * @param context the current action context
49       * @return the dispatcher
50       * @throws Exception if creation fails
51       * @see ClassUtils#getApplicationInstance(String)
52       */
53      protected Dispatcher createDispatcher(String type, ActionContext context) throws Exception {
54          log.info("Initializing dispatcher of type: " + type);
55          return (Dispatcher) ClassUtils.getApplicationInstance(type);
56      }
57  
58      public boolean execute(ActionContext context) throws Exception {
59          // Skip processing if the current request is not valid
60          Boolean valid = context.getFormValid();
61          if ((valid == null) || !valid.booleanValue()) {
62              return CONTINUE_PROCESSING;
63          }
64  
65          // Skip processing if no action is specified
66          if (context.getAction() == null) {
67              return CONTINUE_PROCESSING;
68          }
69  
70          // Skip processing if no dispatcher type specified
71          String dispatcherType = getDispatcherType(context);
72          if (dispatcherType == null) {
73              dispatcherType = defaultDispatcherType;
74              if (dispatcherType == null) {
75                  return CONTINUE_PROCESSING;
76              }
77          }
78  
79          // Obtain (or create) the dispatcher cache
80          String cacheKey = Constants.DISPATCHERS_KEY + context.getModuleConfig().getPrefix();
81          @SuppressWarnings("unchecked")
82          Map<String, Dispatcher> dispatchers = (Map<String, Dispatcher>) context.getApplicationScope().get(cacheKey);
83          if (dispatchers == null) {
84              dispatchers = new HashMap<>();
85              context.getApplicationScope().put(cacheKey, dispatchers);
86          }
87  
88          // Lookup (or create) the dispatch instance
89          Dispatcher dispatcher = null;
90          synchronized (dispatchers) {
91              ActionConfig actionConfig = context.getActionConfig();
92              String actionType = actionConfig.getType();
93              dispatcher = dispatchers.get(actionType);
94  
95              if (dispatcher == null) {
96                  dispatcher = createDispatcher(dispatcherType, context);
97                  dispatchers.put(actionType, dispatcher);
98              }
99          }
100 
101         // Dispatch
102         Object result = dispatcher.dispatch(context);
103         processDispatchResult(result, context);
104 
105         return CONTINUE_PROCESSING;
106     }
107 
108     /**
109      * Retrieves the fully-qualified dispatcher class name to instantiate when
110      * an action mapping does not specify a dispatcher.
111      *
112      * @return the default dispatcher type (fully-qualified class name)
113      * @see #getDispatcherType(ActionContext)
114      * @see #setDefaultDispatcherType(String)
115      */
116     public final String getDefaultDispatcherType() {
117         return defaultDispatcherType;
118     }
119 
120     /**
121      * Retrieves the fully-qualified class name of the dispatcher to instantiate
122      * for the specified context.
123      *
124      * @param context the current action context
125      * @return the class name or <code>null</code>
126      * @see #getDefaultDispatcherType()
127      */
128     protected String getDispatcherType(ActionContext context) {
129         String dispatcherType = null;
130         if (context != null) {
131             dispatcherType = context.getActionConfig().getDispatcher();
132         }
133         return dispatcherType;
134     }
135 
136     /**
137      * Interprets the specified dispatch result. Subclasses should override this
138      * method to provide custom result type handling. Four handlings are
139      * automatically provided:
140      * <ol>
141      * <li><code>null</code> type means the response was handled directly,
142      * and therefore no extra processing is perform</li>
143      * <li>{@link ForwardConfig} type is the classical response, and will be
144      * stored in the context</li>
145      * <li>{@link String} type represents the name of a required forward to
146      * lookup, and will be stored in the context</li>
147      * <li>{@link Void} type means the method had no return signature, and
148      * select the {@link Action#SUCCESS} action forward and store in the context</li>
149      * </ol>
150      *
151      * @param result the result value
152      * @param context the current action context
153      * @throws IllegalStateException if unknown result type or the forward
154      *         cannot be found
155      * @see ActionMapping#findRequiredForward(String)
156      */
157     protected void processDispatchResult(Object result, ActionContext context) {
158         // Null means the response was handled directly
159         if (result == null) {
160             return;
161         }
162 
163         // A forward is the classical response
164         if (result instanceof ForwardConfig) {
165             context.setForwardConfig((ForwardConfig) result);
166             return;
167         }
168 
169         // String represents the name of a forward
170         ActionConfig actionConfig = context.getActionConfig();
171         ActionMapping mapping = ((ActionMapping) actionConfig);
172         if (result instanceof String) {
173             context.setForwardConfig(mapping.findRequiredForward((String) result));
174             return;
175         }
176 
177         // Select success if no return signature
178         if (result == void.class) {
179             context.setForwardConfig(mapping.findRequiredForward(Action.SUCCESS));
180             return;
181         }
182 
183         // Unknown result type
184         throw new IllegalStateException("Unknown dispatch return type: " + result.getClass().getName());
185     }
186 
187     /**
188      * Stores the optional default dispatcher type (fully-qualified class name).
189      *
190      * @param defaultDispatcherType the dispatcher type or <code>null</code>
191      * @see #getDefaultDispatcherType()
192      */
193     public final void setDefaultDispatcherType(String defaultDispatcherType) {
194         this.defaultDispatcherType = defaultDispatcherType;
195     }
196 
197 }
