21  package org.apache.struts.action;
22  
23  import org.apache.commons.logging.Log;
24  import org.apache.commons.logging.LogFactory;
25  import org.apache.struts.config.ActionConfig;
26  import org.apache.struts.config.ControllerConfig;
27  import org.apache.struts.config.ForwardConfig;
28  
29  import java.util.Arrays;
30  
31  /**
32   * <p>An <strong>ActionMapping</strong> represents the information that the
33   * controller, <code>RequestProcessor</code>, knows about the mapping of a
34   * particular request to an instance of a particular <code>Action</code>
35   * class. The <code>ActionMapping</code> instance used to select a particular
36   * <code>Action</code> is passed on to that <code>Action</code>, thereby
37   * providing access to any custom configuration information included with the
38   * <code>ActionMapping</code> object.</p>
39   *
40   * <p>Since Struts 1.1 this class extends <code>ActionConfig</code>.
41   *
42   * <p><strong>NOTE</strong> - This class would have been deprecated and
43   * replaced by <code>org.apache.struts.config.ActionConfig</code> except for
44   * the fact that it is part of the public API that existing applications are
45   * using.</p>
46   *
47   * @version $Rev$ $Date: 2005-08-26 21:58:39 -0400 (Fri, 26 Aug 2005)
48   *          $
49   */
50  public class ActionMapping extends ActionConfig {
51      private static final long serialVersionUID = 5152639565031916976L;
52  
53      /**
54       * <p>Commons Logging instance.</p>
55       *
56       * @since Struts 1.2.8
57       */
58      private static Log log = LogFactory.getLog(ActionMapping.class);
59  
60      /**
61       * <p>Find and return the <code>ForwardConfig</code> instance defining how
62       * forwarding to the specified logical name should be handled. This is
63       * performed by checking local and then global configurations for the
64       * specified forwarding configuration. If no forwarding configuration can
65       * be found, return <code>null</code>.</p>
66       *
67       * @param forwardName Logical name of the forwarding instance to be
68       *                    returned
69       * @return The local or global forward with the specified name.
70       * @see #findRequiredForward(String)
71       */
72      public ActionForward findForward(String forwardName) {
73          ForwardConfig config = findForwardConfig(forwardName);
74  
75          if (config == null) {
76              config = getModuleConfig().findForwardConfig(forwardName);
77          }
78  
79          // TODO: remove warning since findRequiredForward takes care of use case?
80          if (config == null) {
81              if (log.isWarnEnabled()) {
82                  log.warn("Unable to find '" + forwardName + "' forward.");
83              }
84          }
85  
86          return ((ActionForward) config);
87      }
88  
89      /**
90       * <p>Find and return the <code>ForwardConfig</code> instance of this
91       * mapping, throwing an exception if not found locally or globally.</p>
92       *
93       * @param forwardName Logical name of the forwarding instance to be
94       *                    returned
95       * @return The local or global forward with the specified name.
96       * @throws IllegalStateException if the forward is not found
97       * @see #findForward(String)
98       * @since Struts 1.4
99       */
100     public ActionForward findRequiredForward(String forwardName) {
101         ActionForward forward = findForward(forwardName);
102         if (forward == null) {
103             throw new IllegalStateException(
104                     "Unable to find '" + forwardName +
105                     "' forward of action path '" + getPath() + "'");
106         }
107         return forward;
108     }
109 
110     /**
111      * <p>Return the logical names of all locally defined forwards for this
112      * mapping. If there are no such forwards, a zero-length array is
113      * returned.</p>
114      *
115      * @return The forward names for this action mapping.
116      */
117     public String[] findForwards() {
118         ForwardConfig[] fcs = findForwardConfigs();
119         String[] results = new String[fcs.length];
120 
121         Arrays.setAll(results, i -> fcs[i].getName());
122 
123         return results;
124     }
125 
126     /**
127      * <p>Create (if necessary) and return an {@link ActionForward} that
128      * corresponds to the <code>input</code> property of this Action.
129      * <p>
130      * <b>Since Struts 1.4: </b>
131      * If the <code>input</code> property is not specified and the Controller
132      * is configured to interpret the property as a forward, return the
133      * forward named "input" (if it exists) in this action mapping.</p>
134      *
135      * @return The input forward for this action mapping.
136      * @see ControllerConfig#getInputForward()
137      * @since Struts 1.1
138      */
139     public ActionForward getInputForward() {
140         String input = getInput();
141         if (getModuleConfig().getControllerConfig().getInputForward()) {
142             if (input != null) {
143                 return findForward(input);
144             }
145             return findForward(Action.INPUT);
146         }
147         return (new ActionForward(input));
148     }
149 }
