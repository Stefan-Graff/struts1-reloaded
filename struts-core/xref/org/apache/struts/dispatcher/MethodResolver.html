21  package org.apache.struts.dispatcher;
22  
23  import java.io.Serializable;
24  import java.lang.reflect.Method;
25  
26  import org.apache.struts.chain.contexts.ActionContext;
27  
28  /**
29   * This interface defines a pluggable strategy for resolving a method and
30   * extracting its methods from an {@link ActionContext}. A {@link Dispatcher}
31   * may use the resolver to separate out its method resolution based on
32   * technology (such as servlets, portlets, etc.).
33   *
34   * @version $Rev$
35   * @since Struts 1.4
36   */
37  public interface MethodResolver extends Serializable{
38  
39      /**
40       * Constructs the arguments that will be passed to the dispatched method.
41       *
42       * @param context the current action context
43       * @param method the target method of this dispatch
44       *
45       * @return the arguments array for the method, or <code>null</code> if the
46       *         method does not match any supported signatures
47       * @see #resolveMethod(ActionContext, String)
48       */
49      Object[] buildArguments(ActionContext context, Method method);
50  
51      /**
52       * Decides the appropriate method instance for the specified method name.
53       * Implementations may introspect for any desired method signature.
54       *
55       * @param context the current action context
56       * @param methodName the method name to use for introspection
57       * @return the method to invoke
58       * @throws NoSuchMethodException if an appropriate method cannot be found
59       */
60      Method resolveMethod(ActionContext context, String methodName) throws NoSuchMethodException;
61  }
