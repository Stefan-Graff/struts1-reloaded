24  import org.apache.struts.chain.contexts.ActionContext;
25  import org.apache.struts.config.ActionConfig;
26  import org.apache.struts.util.MessageResources;

28  import java.io.Serializable;
29  import java.lang.reflect.InvocationTargetException;
30  import java.lang.reflect.Method;
31  import java.util.HashMap;
32  
33  import org.apache.commons.logging.Log;
34  import org.apache.commons.logging.LogFactory;

36  /**
37   * This abstract class is the stock template for {@link Dispatcher}
38   * implementations.
39   *
40   * @version $Rev$
41   * @since Struts 1.4
42   */
43  public abstract class AbstractDispatcher implements Dispatcher, Serializable {
44      private static final long serialVersionUID = 2259767062109056338L;
45  
46      // Package message bundle keys
47      static final String LOCAL_STRINGS = "org.apache.struts.dispatcher.LocalStrings";
48      static final String MSG_KEY_DISPATCH_ERROR = "dispatcher.error";
49      static final String MSG_KEY_MISSING_METHOD = "dispatcher.missingMethod";
50      static final String MSG_KEY_MISSING_METHOD_LOG = "dispatcher.missingMethod.log";
51      static final String MSG_KEY_MISSING_MAPPING_PARAMETER = "dispatcher.missingMappingParameter";
52      static final String MSG_KEY_UNSPECIFIED = "dispatcher.unspecified";
53  
54      /**
55       * The name of the <code>cancelled</code> method.
56       *
57       * @see ActionContext#getCancelled()
58       */
59      public static final String CANCELLED_METHOD_NAME = "cancelled";
60  
61      /**
62       * The name of the <code>execute</code> method.
63       */
64      public static final String EXECUTE_METHOD_NAME = "execute";
65  
66      /**
67       * The message resources for this package.
68       */
69      static MessageResources messages = MessageResources.getMessageResources(LOCAL_STRINGS);
70  
71      /**
72       * Shared commons Logging instance among subclasses.
73       */
74      protected transient final Log log;
75  
76      /**
77       * The dictionary of {@link Method} objects we have introspected for this
78       * class, keyed by method name. This collection is populated as different
79       * methods are called, so that introspection needs to occur only once per
80       * method name.
81       */
82      private transient final HashMap<String, Method> methods;
83  
84      private final MethodResolver methodResolver;
85  
86      /**
87       * Constructs a new dispatcher with the specified method resolver.
88       *
89       * @param methodResolver the method resolver
90       */
91      public AbstractDispatcher(MethodResolver methodResolver) {
92          this.methodResolver = methodResolver;
93          log = LogFactory.getLog(getClass());
94          methods = new HashMap<>();
95      }
96  
97      /**
98       * Constructs the arguments that will be passed to the dispatched method.
99       * The construction is delegated to the method resolver instance.
100      *
101      * @param context the current action context
102      * @param method the target method of this dispatch
103      * @return the arguments array
104      * @throws IllegalStateException if the method does not have a supported
105      *         signature
106      * @see MethodResolver#buildArguments(ActionContext, Method)
107      */
108     Object[] buildMethodArguments(ActionContext context, Method method) {
109         Object[] args = methodResolver.buildArguments(context, method);
110         if (args == null) {
111             throw new IllegalStateException("Unsupported method signature: " + method.toString());
112         }
113         return args;
114     }
115 
116     public Object dispatch(ActionContext context) throws Exception {
117         // Resolve the method name; fallback to default if necessary
118         String methodName = resolveMethodName(context);
119         if ((methodName == null) || "".equals(methodName)) {
120             methodName = getDefaultMethodName();
121         }
122 
123         // Ensure there is a specified method name to invoke.
124         // This may be null if the user hacks the query string.
125         if (methodName == null) {
126             return unspecified(context);
127         }
128 
129         // Identify the method object to dispatch
130         Method method;
131         try {
132             method = getMethod(context, methodName);
133         } catch (NoSuchMethodException e) {
134             // The log message reveals the offending method name...
135             String path = context.getActionConfig().getPath();
136             String message = messages.getMessage(MSG_KEY_MISSING_METHOD_LOG, path, methodName);
137             log.error(message, e);
138 
139             // ...but the exception thrown does not
140             // See r383718 (XSS vulnerability)
141             String userMsg = messages.getMessage(MSG_KEY_MISSING_METHOD, path);
142             NoSuchMethodException e2 = new NoSuchMethodException(userMsg);
143             e2.initCause(e);
144             throw e2;
145         }
146 
147         // Invoke the named method and return its result
148         return dispatchMethod(context, method, methodName);
149     }
150 
151     /**
152      * Dispatches to the specified method.
153      *
154      * @param context the current action context
155      * @param method The method to invoke
156      * @param name The name of the method to invoke
157      * @return the return value of the method
158      * @throws Exception if the dispatch fails with an exception
159      * @see #buildMethodArguments(ActionContext, Method)
160      */
161     protected final Object dispatchMethod(ActionContext context, Method method, String name) throws Exception {
162         Action target = context.getAction();
163         String path = context.getActionConfig().getPath();
164         Object[] args = buildMethodArguments(context, method);
165         return invoke(target, method, args, path);
166     }
167 
168     /**
169      * Empties the method cache.
170      *
171      * @see #getMethod(ActionContext, String)
172      */
173     final void flushMethodCache() {
174         synchronized (methods) {
175             methods.clear();
176         }
177     }
178 
179     /**
180      * Retrieves the name of the method to fallback upon if no method name can
181      * be resolved. The default implementation returns
182      * {@link #EXECUTE_METHOD_NAME}.
183      *
184      * @return the fallback method name; can be <code>null</code>
185      * @see #resolveMethodName(ActionContext)
186      * @see #EXECUTE_METHOD_NAME
187      */
188     protected String getDefaultMethodName() {
189         return EXECUTE_METHOD_NAME;
190     }
191 
192     /**
193      * Introspects the action to identify a method of the specified name that
194      * will be the target of the dispatch. This implementation caches the method
195      * instance for subsequent invocations.
196      *
197      * @param context the current action context
198      * @param methodName the name of the method to be introspected
199      * @return the method of the specified name
200      * @throws NoSuchMethodException if no such method can be found
201      * @see #resolveMethod(ActionContext, String)
202      * @see #flushMethodCache()
203      */
204     protected final Method getMethod(ActionContext context, String methodName) throws NoSuchMethodException {
205         synchronized (methods) {
206             // Key the method based on the class-method combination
207             StringBuilder keyBuf = new StringBuilder(100);
208             keyBuf.append(context.getAction().getClass().getName());
209             keyBuf.append(":");
210             keyBuf.append(methodName);
211             String key = keyBuf.toString();
212 
213             Method method = methods.get(key);
214 
215             if (method == null) {
216                 method = resolveMethod(context, methodName);
217                 methods.put(key, method);
218             }
219 
220             return method;
221         }
222     }
223 
224     /**
225      * Convenience method to help dispatch the specified method. The method is
226      * invoked via reflection.
227      *
228      * @param target the target object
229      * @param method the method of the target object
230      * @param args the arguments for the method
231      * @param path the mapping path
232      * @return the return value of the method
233      * @throws Exception if the dispatch fails with an exception
234      */
235     protected final Object invoke(Object target, Method method, Object[] args, String path) throws Exception {
236         try {
237             Object retval = method.invoke(target, args);
238             if (method.getReturnType() == void.class) {
239                 retval = void.class;
240             }
241             return retval;
242         } catch (IllegalAccessException e) {
243             String message = messages.getMessage(MSG_KEY_DISPATCH_ERROR, path);
244             log.error(message + ":" + method.getName(), e);
245             throw e;
246         } catch (InvocationTargetException e) {
247             // Rethrow the target exception if possible so that the
248             // exception handling machinery can deal with it
249             Throwable t = e.getTargetException();
250             if (t instanceof Exception) {
251                 throw (Exception) t;
252             } else {
253                 String message = messages.getMessage(MSG_KEY_DISPATCH_ERROR, path);
254                 log.error(message + ":" + method.getName(), e);
255                 throw new Exception(t);
256             }
257         }
258     }
259 
260     /**
261      * Determines whether the current form's cancel button was pressed. The
262      * default behavior method will check if the
263      * {@link ActionContext#getCancelled()} context property is set , which
264      * normally occurs if the cancel button generated by <strong>CancelTag</strong>
265      * was pressed by the user in the current request.
266      *
267      * @param context the current action context
268      * @return <code>true</code> if the request is cancelled; otherwise
269      *         <code>false</code>
270      */
271     protected boolean isCancelled(ActionContext context) {
272         Boolean cancelled = context.getCancelled();
273         return (cancelled != null) && cancelled.booleanValue();
274     }
275 
276     /**
277      * Decides the appropriate method instance for the specified method name.
278      * Implementations may introspect for any desired method signature. This
279      * resolution is only invoked if {@link #getMethod(ActionContext, String)}
280      * does not find a match in its method cache.
281      *
282      * @param context the current action context
283      * @param methodName the method name to use for introspection
284      * @return the method to invoke
285      * @throws NoSuchMethodException if an appropriate method cannot be found
286      * @see #getMethod(ActionContext, String)
287      * @see #invoke(Object, Method, Object[], String)
288      */
289     Method resolveMethod(ActionContext context, String methodName) throws NoSuchMethodException {
290         return methodResolver.resolveMethod(context, methodName);
291     }
292 
293     /**
294      * Decides the method name that can handle the request.
295      *
296      * @param context the current action context
297      * @return the method name or <code>null</code> if no name can be resolved
298      * @see #getDefaultMethodName()
299      * @see #resolveMethod(ActionContext, String)
300      */
301     abstract String resolveMethodName(ActionContext context);
302 
303     /**
304      * Services the case when the dispatch fails because the method name cannot
305      * be resolved. The default behavior throws an {@link IllegalStateException}.
306      * Subclasses should override this to provide custom handling such as
307      * sending an HTTP 404 error or dispatching elsewhere.
308      *
309      * @param context the current action context
310      * @return the return value of the dispatch
311      * @throws Exception if an error occurs
312      * @see #resolveMethodName(ActionContext)
313      */
314     protected Object unspecified(ActionContext context) throws Exception {
315         ActionConfig config = context.getActionConfig();
316         String msg = messages.getMessage(MSG_KEY_UNSPECIFIED, config.getPath());
317         log.error(msg);
318         throw new IllegalStateException(msg);
319     }
