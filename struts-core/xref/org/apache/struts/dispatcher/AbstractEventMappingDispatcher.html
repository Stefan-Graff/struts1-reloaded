21  package org.apache.struts.dispatcher;
22  
23  import org.apache.struts.action.ActionMapping;
24  import org.apache.struts.chain.contexts.ActionContext;
25  
26  import java.util.StringTokenizer;
27  
28  /**
29   * This abstract class is a template for choosing the target method that is
30   * named in the list-encoded <code>parameter</code> attribute of the
31   * {@link ActionMapping} that matches a submission parameter. This is useful for
32   * developers who prefer to use many submit buttons, images, or submit links on
33   * a single form and whose related actions exist in a single action.
34   * <p>
35   * For utility purposes, you can use the <code>key=value</code> notation to
36   * alias methods so that they are exposed as different form element names, in
37   * the event of a naming conflict or otherwise. In this example, the
38   * <em>recalc</em> button (via a submission parameter) will invoke the
39   * <code>recalculate</code> method. The security-minded person may find this
40   * feature valuable to obfuscate and not expose the methods.
41   * <p>
42   * The <em>default</em> key is purely optional, but encouraged when nothing
43   * matches (such as the user pressing the enter key). If this is not specified
44   * and no parameters match the list of method keys, the method resolution
45   * returns <code>null</code>.
46   * <p>
47   * The order of the parameters are guaranteed to be iterated in the order
48   * specified. If multiple buttons were accidently submitted, the first match in
49   * the list will be dispatched.
50   *
51   * @version $Rev$
52   * @since Struts 1.4
53   */
54  public abstract class AbstractEventMappingDispatcher extends AbstractMappingDispatcher {
55  
56      /**
57       * The method key, if present, to use if other specified method keys do not
58       * match a request parameter.
59       */
60      protected static final String DEFAULT_METHOD_KEY = "default";
61  
62      /**
63       * Constructs a new dispatcher with the specified method resolver.
64       *
65       * @param methodResolver the method resolver
66       */
67      public AbstractEventMappingDispatcher(MethodResolver methodResolver) {
68          super(methodResolver);
69      }
70  
71      /**
72       * Determines whether the specified method key is a submission parameter.
73       *
74       * @param context the current action context
75       * @param methodKey the method key
76       * @return <code>true</code> if match; otherwise <code>false</code>
77       * @see #resolveMethodName(ActionContext)
78       */
79      protected abstract boolean isSubmissionParameter(ActionContext context, String methodKey);
80  
81      protected final String resolveMethodName(ActionContext context) {
82          // Obtain the mapping parameter
83          String mappingParameter = super.resolveMethodName(context);
84          if (mappingParameter == null) {
85              return null;
86          }
87  
88          // Parse it as a comma-separated list
89          StringTokenizer st = new StringTokenizer(mappingParameter, ",");
90          String defaultMethodName = null;
91  
92          while (st.hasMoreTokens()) {
93              String methodKey = st.nextToken().trim();
94              String methodName = methodKey;
95  
96              // The key can either be a direct method name or an alias
97              // to a method as indicated by a "key=value" signature
98              int equals = methodKey.indexOf('=');
99              if (equals > -1) {
100                 methodName = methodKey.substring(equals + 1).trim();
101                 methodKey = methodKey.substring(0, equals).trim();
102             }
103 
104             // Set the default if it passes by
105             if (methodKey.equals(DEFAULT_METHOD_KEY)) {
106                 defaultMethodName = methodName;
107             }
108 
109             // Is it a match?
110             if (isSubmissionParameter(context, methodKey)) {
111                 return methodName;
112             }
113         }
114 
115         return defaultMethodName;
116     }
117 
118 }
