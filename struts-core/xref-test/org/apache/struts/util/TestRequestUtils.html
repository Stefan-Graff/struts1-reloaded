245         // Retrieve an appropriately configured DynaActionForm instance
246         request.setPathElements("/myapp", "/dynamic0.do", null, null);

257         ActionForm form =
258             RequestUtils.createActionForm(request, mapping, moduleConfig, null);
259 
260         assertNotNull(form, "ActionForm returned");
261         assertInstanceOf(DynaActionForm.class, form, "ActionForm of correct type");
262 
263         // Validate the property values
264         DynaActionForm dform = (DynaActionForm) form;
265         Boolean booleanProperty = (Boolean) dform.get("booleanProperty");
266 
267         assertTrue(booleanProperty.booleanValue(), "booleanProperty is true");
268 
269         String stringProperty = (String) dform.get("stringProperty");
270 
271         assertEquals("String Property", stringProperty,
272             "stringProperty is correct");
273 
274         Object value = null;
275 
276         value = dform.get("intArray1");
277         assertNotNull(value, "intArray1 exists");
278         assertInstanceOf(int[].class, value, "intArray1 is int[]");
279 
280         int[] intArray1 = (int[]) value;
281 
282         assertEquals(3, intArray1.length, "intArray1 length is correct");
283         assertEquals(1, intArray1[0], "intArray1[0] value is correct");
284         assertEquals(2, intArray1[1], "intArray1[1] value is correct");
285         assertEquals(3, intArray1[2], "intArray1[2] value is correct");
286 
287         value = dform.get("intArray2");
288         assertNotNull(value, "intArray2 exists");
289         assertInstanceOf(int[].class, value, "intArray2 is int[]");
290 
291         int[] intArray2 = (int[]) value;
292 
293         assertEquals(5, intArray2.length, "intArray2 length is correct");
294         assertEquals(0, intArray2[0], "intArray2[0] value is correct");
295         assertEquals(0, intArray2[1], "intArray2[1] value is correct");
296         assertEquals(0, intArray2[2], "intArray2[2] value is correct");
297         assertEquals(0, intArray2[3], "intArray2[3] value is correct");
298         assertEquals(0, intArray2[4], "intArray2[4] value is correct");
299 
300         value = dform.get("principal");
301         assertNotNull(value, "principal exists");
302         assertInstanceOf(MockPrincipal.class, value, "principal is MockPrincipal");
303 
304         value = dform.get("stringArray1");
305         assertNotNull(value, "stringArray1 exists");
306         assertInstanceOf(String[].class, value, "stringArray1 is int[]");
307 
308         String[] stringArray1 = (String[]) value;
309 
310         assertEquals(3, stringArray1.length, "stringArray1 length is correct");
311         assertEquals("aaa", stringArray1[0], "stringArray1[0] value is correct");
312         assertEquals("bbb", stringArray1[1], "stringArray1[1] value is correct");
313         assertEquals("ccc", stringArray1[2], "stringArray1[2] value is correct");
314 
315         value = dform.get("stringArray2");
316         assertNotNull(value, "stringArray2 exists");
317         assertInstanceOf(String[].class, value, "stringArray2 is int[]");
318 
319         String[] stringArray2 = (String[]) value;
320 
321         assertEquals(3, stringArray2.length, "stringArray2 length is correct");
322         assertEquals(new String(), stringArray2[0],
323             "stringArray2[0] value is correct");
324         assertEquals(new String(), stringArray2[1],
325             "stringArray2[1] value is correct");
326         assertEquals(new String(), stringArray2[2],
327             "stringArray2[2] value is correct");
328 
329         // Different form beans should get different property value instances
330         Object value1 = null;
331         DynaActionForm dform1 =
332             (DynaActionForm) RequestUtils.createActionForm(request, mapping,
333                 moduleConfig, null);
334 
335         value = dform.get("principal");
336         value1 = dform1.get("principal");
337         assertEquals(value, value1,
338             "Different form beans get equal instance values");
339         assertNotSame(value, value1,
340             "Different form beans get different instances 1");
341 
342         value = dform.get("stringArray1");
343         value1 = dform1.get("stringArray1");
344         assertNotSame(value, value1,
345             "Different form beans get different instances 2");
346 
347         dform1.set("stringProperty", "Different stringProperty value");
348         value = dform.get("stringProperty");
349         value1 = dform1.get("stringProperty");
350         assertNotSame(value, value1,
351             "Different form beans get different instances 3");
