70      /**
71       * Set up instance variables required by this test case.
72       */
73      @BeforeEach
74      public void setUp() {
75          ModuleConfigFactory factoryObject = ModuleConfigFactory.createFactory();
76  
77          moduleConfig = factoryObject.createModuleConfig("");
78  
79          // Setup the base and sub forwards, with sub extending base
80          baseConfig = new ForwardConfig();
81          baseConfig.setName("baseConfig");
82          baseConfig.setPath("/somePage.jsp");
83  
84          subConfig = new ForwardConfig();
85          subConfig.setName("subConfig");
86          subConfig.setExtends("baseConfig");
87          subConfig.setRedirect(true);
88  
89          actionBaseConfig = new ForwardConfig();
90          actionBaseConfig.setName("baseConfig");
91          actionBaseConfig.setExtends("baseConfig");
92          actionBaseConfig.setModule("/other");
93  
94          // Setup the default action config
95          actionConfig = new ActionConfig();
96          actionConfig.setPath("/index");
97          moduleConfig.addActionConfig(actionConfig);
98  
99          // No forward configs are registered to either module or action configs.
100         // Each test will determine where it needs these configs, if at all.
101     }

104      * Tear down instance variables required by this test case.

260     /**
261      * Test processExtends() when nothing is extended.
262      */
263     @Test
264     public void testProcessExtendsNoExtension()
265         throws Exception {
266         String path = baseConfig.getPath();
267         String module = baseConfig.getModule();
268         String name = baseConfig.getName();
269         String inherit = baseConfig.getExtends();
270         boolean redirect = baseConfig.getRedirect();
271 
272         moduleConfig.addForwardConfig(baseConfig);
273         baseConfig.processExtends(moduleConfig, null);
274 
275         assertEquals(path, baseConfig.getPath(), "Path shouldn't have changed");
276         assertEquals(module, baseConfig.getModule(),
277             "Module shouldn't have changed");
278         assertEquals(name, baseConfig.getName(), "Name shouldn't have changed");
279         assertEquals(inherit, baseConfig.getExtends(),
280             "Extends shouldn't have changed");
281         assertEquals(redirect, baseConfig.getRedirect(),
282             "Redirect shouldn't have changed");
283     }

285     /**
286      * Test processExtends() with a basic extension.
287      */
288     @Test
289     public void testProcessExtendsBasicExtension()
290         throws Exception {
291         String baseCount = "10";
292 
293         baseConfig.setProperty("count", baseCount);
294 
295         String baseLabel = "label a";
296 
297         baseConfig.setProperty("label", baseLabel);
298 
299         String path = baseConfig.getPath();
300         String module = baseConfig.getModule();
301 
302         String inherit = subConfig.getExtends();
303         String name = subConfig.getName();
304         boolean redirect = subConfig.getRedirect();
305 
306         String subLabel = "label b";
307 
308         subConfig.setProperty("label", subLabel);
309 
310         moduleConfig.addForwardConfig(baseConfig);
311         moduleConfig.addForwardConfig(subConfig);
312         subConfig.processExtends(moduleConfig, null);
313 
314         assertEquals(path, subConfig.getPath(), "Path wasn't inherited");
315         assertEquals(module, subConfig.getModule(), "Module wasn't inherited");
316         assertEquals(name, subConfig.getName(), "Name shouldn't have changed");
317         assertEquals(inherit, subConfig.getExtends(),
318             "Extends shouldn't have changed");
319         assertEquals(redirect, subConfig.getRedirect(),
320             "Redirect shouldn't have changed");
321         assertEquals(baseCount, subConfig.getProperty("count"),
322             "Arbitrary config property was not inherited");
323         assertEquals(subLabel, subConfig.getProperty("label"),
324             "Overridden config property was not retained");
325     }

400      * ForwardConfig, which in turn extends a global ForwardConfig with the

403     @Test
404     public void testProcessExtendsActionExtendsActionExtendsGlobalWithSameName()
405         throws Exception {
406         String path = baseConfig.getPath();
407 
408         String module = actionBaseConfig.getModule();
409 
410         boolean redirect = subConfig.getRedirect();
411         String inherit = subConfig.getExtends();

413 
414         moduleConfig.addForwardConfig(baseConfig);
415         actionConfig.addForwardConfig(actionBaseConfig);
416         actionConfig.addForwardConfig(subConfig);
417 
418         subConfig.processExtends(moduleConfig, actionConfig);
419 
420         assertTrue(baseConfig.extensionProcessed,
421             "baseConfig's processExtends() should've been called");

423             "actionBaseConfig's processExtends() should've been called");

426         assertEquals(module, subConfig.getModule(), "Module wasn't inherited");
