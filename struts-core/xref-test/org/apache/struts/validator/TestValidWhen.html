286     /**
287      * Parse the expression and check that the expected result (either true or
288      * false) occurs - fail if an exception is thrown opr the wrong result
289      * occurs.
290      *
291      * @param test     Test expression
292      * @param bean     Test Bean
293      * @param index    index value
294      * @param property Bean property
295      * @param expected Expected Result
296      */
297     private void doParse(String test, Object bean, int index, String property,
298         boolean expected) {
299         boolean result = false;
300 
301         try {
302 //            System.out.println("---------------------------");
303 //            System.out.println(test + " " + expected);
304             result = doParse(test, bean, index, property);
305         } catch (Exception ex) {
306 //            System.out.println("Exception: " + ex);
307 //            ex.printStackTrace();
308             log.error("Parsing {} for property '{}'", test, property, ex);
309             fail("Parsing " + test + " threw " + ex);
310         }
311 
312         if (expected) {
313             assertTrue(result, test + " didn't return TRUE for " + property);
314         } else {
315             assertFalse(result, test + " didn't return FALSE for " + property);
316         }
317     }

342     /**
343      * Parse the expression returning the result
344      *
345      * @param test     Test expression
346      * @param bean     Test Bean
347      * @param index    index value
348      * @param property Bean property
349      */
350     private boolean doParse(String test, Object bean, int index, String property)
351         throws Exception {
352         if (bean == null) {
353             throw new NullPointerException("Bean is null for property '"
354                 + property + "'");
355         }
356 
357         String value =
358             String.class.isInstance(bean) ? (String) bean
359                                           : ValidatorUtils.getValueAsString(bean,
360                 property);
361 
362         ValidWhenLexer lexer = new ValidWhenLexer(CharStreams.fromString(test));
363 
364         ValidWhenParser parser = new ValidWhenParser(new CommonTokenStream(lexer));
365 
366         ValidWhenEvaluator validWhenEvaluator = new ValidWhenEvaluator(bean, value, index);
367 
368         ExpressionContext expressionContext = parser.expression();
369         ValidWhenResult<?> result = validWhenEvaluator.visitExpression(expressionContext);
370 
371         return result == null ? false : result.toBoolean();
372     }
