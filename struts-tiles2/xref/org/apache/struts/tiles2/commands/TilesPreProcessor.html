21  package org.apache.struts.tiles2.commands;
22  
23  import org.apache.commons.chain.Command;
24  import org.apache.commons.chain.Context;
25  import org.apache.struts.chain.contexts.ServletActionContext;
26  import org.apache.struts.config.ForwardConfig;
27  import org.apache.tiles.TilesContainer;
28  import org.apache.tiles.access.TilesAccess;
29  import org.apache.tiles.request.ApplicationContext;
30  import org.apache.tiles.request.Request;
31  import org.apache.tiles.request.servlet.ServletRequest;
32  import org.apache.tiles.request.servlet.ServletUtil;
33  import org.slf4j.Logger;
34  import org.slf4j.LoggerFactory;
35  
36  
37  /**
38   * <p>Command class intended to perform responsibilities of the
39   * TilesRequestProcessor in Struts 1.1.  Does not actually dispatch requests,
40   * but simply prepares the chain context for a later forward as
41   * appropriate.  Should be added to a chain before something which
42   * would handle a conventional ForwardConfig.</p>
43   *
44   * <p>This class will never have any effect on the chain unless a
45   * <code>TilesDefinitionFactory</code> can be found; however it does not
46   * consider the absence of a definition factory to be a fatal error; the
47   * command simply returns false and lets the chain continue.</p>
48   *
49   * <p>To initialize the <code>TilesDefinitionFactory</code>, use
50   * <code>org.apache.struts.chain.commands.legacy.TilesPlugin</code>.  This class
51   * is a simple extension to <code>org.apache.struts.tiles2.TilesPlugin</code>
52   * which simply does not interfere with your choice of <code>RequestProcessor</code>
53   * implementation.
54   *  </p>
55   *
56   *
57   */
58  public class TilesPreProcessor implements Command {
59  
60  
61      // ------------------------------------------------------ Instance Variables
62  
63  
64      /**
65       * The {@code Log} instance for this class.
66       */
67      private final Logger log =
68          LoggerFactory.getLogger(TilesPreProcessor.class);
69  
70      // ---------------------------------------------------------- Public Methods
71  
72  
73      /**
74       * <p>If the current <code>ForwardConfig</code> is using "tiles",
75       * perform necessary pre-processing to set up the <code>TilesContext</code>
76       * and substitute a new <code>ForwardConfig</code> which is understandable
77       * to a <code>RequestDispatcher</code>.</p>
78       *
79       * <p>Note that if the command finds a previously existing
80       * <code>AttributeContext</code> in the request, then it
81       * infers that it has been called from within another tile,
82       * so instead of changing the <code>ForwardConfig</code> in the chain
83       * <code>Context</code>, the command uses <code>RequestDispatcher</code>
84       * to <em>include</em> the tile, and returns true, indicating that the processing
85       * chain is complete.</p>
86       *
87       * @param context The <code>Context</code> for the current request
88       *
89       * @throws Exception If something goes wrong.
90       * @return <code>false</code> in most cases, but true if we determine
91       * that we're processing in "include" mode.
92       */
93      public boolean execute(Context context) throws Exception {
94  
95          // Is there a Tiles Definition to be processed?
96          ServletActionContext sacontext = (ServletActionContext) context;
97          ForwardConfig forwardConfig = sacontext.getForwardConfig();
98          if (forwardConfig == null || forwardConfig.getPath() == null) {
99              // this is not a serious error, so log at low priority
100             log.debug("No forwardConfig or no path, so pass to next command.");
101             return (false);
102         }
103 
104 
105         ApplicationContext applicationContext = ServletUtil
106                 .getApplicationContext(sacontext.getContext());
107         Request request = new ServletRequest(applicationContext,
108                 sacontext.getRequest(), sacontext.getResponse());
109         TilesContainer container = TilesAccess.getContainer(applicationContext);
110         if (container == null) {
111             log.debug("Tiles container not found, so pass to next command.");
112             return false;
113         }
114 
115         if (container.isValidDefinition(forwardConfig.getPath(), request)) {
116             container.render(forwardConfig.getPath(), request);
117             sacontext.setForwardConfig(null);
118         } else {
119             // ignore not found
120             log.debug("Cannot find definition '{}'", forwardConfig.getPath());
121         }
122 
123         return false;
124     }
125 }
