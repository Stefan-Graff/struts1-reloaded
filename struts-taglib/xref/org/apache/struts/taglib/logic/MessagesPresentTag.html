21  package org.apache.struts.taglib.logic;
22  
23  import org.apache.struts.Globals;
24  import org.apache.struts.action.ActionMessage;
25  import org.apache.struts.action.ActionMessages;
26  import org.apache.struts.taglib.TagUtils;
27  
28  import jakarta.servlet.jsp.JspException;
29  
30  import java.util.Iterator;
31  
32  /**
33   * Evaluate to <code>true</code> if an <code>ActionMessages</code> class or a
34   * class that can be converted to an <code>ActionMessages</code> class is in
35   * request scope under the specified key and there is at least one message in
36   * the class or for the property specified.
37   *
38   * @version $Rev$ $Date: 2005-11-19 12:36:20 -0500 (Sat, 19 Nov 2005)
39   *          $
40   * @since Struts 1.1
41   */
42  public class MessagesPresentTag extends ConditionalTagBase {
43      private static final long serialVersionUID = 4848602425179872796L;
44  
45      /**
46       * If this is set to 'true', then the <code>Globals.MESSAGE_KEY</code>
47       * will be used to retrieve the messages from scope.
48       */
49      protected String message = null;
50  
51      /**
52       * The name of the page-scoped attribute to be populated
53       * with the message count of the specifie dproperty.
54       */
55      protected String count;
56  
57      public MessagesPresentTag() {
58          name = Globals.ERROR_KEY;
59      }
60  
61      public String getMessage() {
62          return (this.message);
63      }
64  
65      public void setMessage(String message) {
66          this.message = message;
67      }
68  
69      public void setCount(String count) {
70          this.count = count;
71      }
72  
73      public String getCount() {
74          return count;
75      }
76  
77      /**
78       * Evaluate the condition that is being tested by this particular tag, and
79       * return <code>true</code> if the nested body content of this tag should
80       * be evaluated, or <code>false</code> if it should be skipped.
81       *
82       * @throws JspException if a JSP exception occurs
83       */
84      protected boolean condition()
85          throws JspException {
86          return (condition(true));
87      }
88  
89      /**
90       * Evaluate the condition that is being tested by this particular tag, and
91       * return <code>true</code> if there is at least one message in the class
92       * or for the property specified.
93       *
94       * @param desired Desired outcome for a true result
95       * @throws JspException if a JSP exception occurs
96       */
97      protected boolean condition(boolean desired)
98          throws JspException {
99          ActionMessages am = null;
100 
101         String key = name;
102 
103         if ((message != null) && "true".equalsIgnoreCase(message)) {
104             key = Globals.MESSAGE_KEY;
105         }
106 
107         try {
108             am = TagUtils.getInstance().getActionMessages(pageContext, key);
109         } catch (JspException e) {
110             TagUtils.getInstance().saveException(pageContext, e);
111             throw e;
112         }
113 
114         // Acquire the collection of messages
115         Iterator<ActionMessage> iterator;
116         int size;
117         if (property == null) {
118             iterator = am.get();
119             size = am.size();
120         } else {
121             iterator = am.get(property);
122             size = am.size(property);
123         }
124 
125         // Expose the count when specified
126         if (count != null) {
127             pageContext.setAttribute(count, Integer.valueOf(size));
128         }
129 
130         return (iterator.hasNext() == desired);
131     }
132 
133     public int doEndTag() throws JspException {
134         if (count != null) {
135             pageContext.removeAttribute(count);
136         }
137         return super.doEndTag();
138     }
139 
140     /**
141      * Release all allocated resources.
142      */
143     public void release() {
144         super.release();
145         name = Globals.ERROR_KEY;
146         message = null;
147         count = null;
148     }
149 }
