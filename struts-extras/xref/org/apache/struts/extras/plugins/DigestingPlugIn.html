88      /**
89       * <p>Initialize a <code>Digester</code> and use it to parse a
90       * configuration file, resulting in a root object which will be placed
91       * into the ServletContext.</p>
92       *
93       * @param servlet ActionServlet that is managing all the modules in this
94       *                web application
95       * @param config  ModuleConfig for the module with which this plug-in is
96       *                associated
97       * @throws ServletException if this <code>PlugIn</code> cannot be
98       *                          successfully initialized
99       */
100     public void init(ActionServlet servlet, ModuleConfig config)
101         throws ServletException {
102         this.servlet = servlet;
103         this.moduleConfig = config;
104 
105         Object obj = null;
106 
107         Digester digester = this.initializeDigester();
108 
109         if (this.push) {
110             log.debug("push == true; pushing plugin onto digester stack");
111             digester.push(this);
112         }
113 
114         try {
115             log.debug("XML data file: [path: {}, source: {}]",
116                 this.configPath, this.configSource);
117 
118             URL configURL =
119                 this.getConfigURL(this.configPath, this.configSource);
120 
121             if (configURL == null) {
122                 throw new ServletException(
123                     "Unable to locate XML data file at [path: "
124                     + this.configPath + ", source: " + this.configSource + "]");
125             }
126 
127             URLConnection conn = configURL.openConnection();
128 
129             conn.setUseCaches(false);
130             conn.connect();
131             obj = digester.parse(conn.getInputStream());
132         } catch (IOException e) {
133             // TODO Internationalize msg
134             log.error("Exception processing config", e);
135             throw new ServletException(e);
136         } catch (SAXException e) {
137             // TODO Internationalize msg
138             log.error("Exception processing config", e);
139             throw new ServletException(e);
140         }
141 
142         this.storeGeneratedObject(obj);
143     }

145     /**
146      * Initialize the <code>Digester</code> which will be used to process the
147      * main configuration.
148      *
149      * @return a Digester, ready to use.
150      * @throws ServletException
151      */
152     protected Digester initializeDigester()
153         throws ServletException {
154         Digester digester = null;
155 
156         if ((this.digesterPath != null) && (this.digesterSource != null)) {
157             try {
158                 log.debug("Initialize digester from XML [path: {}; source: {}]",
159                     this.digesterPath, this.digesterSource);
160                 digester =
161                     this.digesterFromXml(this.digesterPath, this.digesterSource);
162             } catch (IOException e) {
163                 // TODO Internationalize msg
164                 log.error("Exception instantiating digester from XML", e);
165                 throw new ServletException(e);
166             }
167         } else {
168             log.debug("No XML rules for digester; call newDigesterInstance()");
169             digester = this.newDigesterInstance();
170         }
171 
172         this.applyRuleSets(digester);
173 
174         return digester;
175     }

179      * override this to provide a subclass of Digester, or to configure the

188     /**
189      * <p>Instantiate a Digester from an XML input stream using the Commons
190      * <code>DigesterLoader</code>.</p>
191      *
192      * @param path   the path to the digester rules XML to be found using
193      *               <code>source</code>
194      * @param source a string indicating the lookup method to be used with
195      *               <code>path</code>
196      * @return a configured Digester
197      * @throws FileNotFoundException
198      * @throws MalformedURLException
199      * @see #getConfigURL(String, String)
200      */
201     protected Digester digesterFromXml(String path, String source)
202         throws IOException {
203         URL configURL = this.getConfigURL(path, source);
204 
205         if (configURL == null) {
206             throw new NullPointerException("No resource '" + path
207                 + "' found in '" + source + "'");
208         }
209 
210 //        Digester3:
211 //        RulesModule rules = new FromXmlRulesModule() {
212 //            @Override
213 //            protected void loadRules() {
214 //                loadXMLRules(configURL);
215 //            }
216 //        };
217 //        return DigesterLoader.newLoader(rules).newDigester();
218         return DigesterLoader.createDigester(configURL);
219     }

221     /**
222      * Instantiate any <code>RuleSet</code> classes defined in the
223      * <code>rulesets</code> property and use them to add rules to our
224      * <code>Digester</code>.
225      *
226      * @param digester the Digester instance to add RuleSet objects to.
227      * @throws ServletException
228      */
229     protected void applyRuleSets(Digester digester)
230         throws ServletException {
231         if ((this.rulesets == null) || (this.rulesets.trim().length() == 0)) {
232             return;
233         }
234 
235         rulesets = rulesets.trim();
236 
237         String ruleSet = null;
238 
239         while (rulesets.length() > 0) {
240             int comma = rulesets.indexOf(",");
241 
242             if (comma < 0) {
243                 ruleSet = rulesets.trim();
244                 rulesets = "";
245             } else {
246                 ruleSet = rulesets.substring(0, comma).trim();
247                 rulesets = rulesets.substring(comma + 1).trim();
248             }
249 
250             // TODO Internationalize msg
251             log.debug("Configuring custom Digester Ruleset of type {}"
252                 , ruleSet);
253 
254             try {
255                 RuleSet instance =
256                     (RuleSet) RequestUtils.applicationInstance(ruleSet);
257 
258                 digester.addRuleSet(instance);
259             } catch (Exception e) {
260                 // TODO Internationalize msg
261                 log.error("Exception configuring custom Digester RuleSet", e);
262                 throw new ServletException(e);
263             }
264         }
265     }

267     /**
268      * <p>Look up a resource path using one of a set of known path resolution
269      * mechanisms and return a URL to the resource.</p>
270      *
271      * @param path   a String which is meaningful to one of the known
272      *               resolution mechanisms.
273      * @param source one of the known path resolution mechanisms:
274      *
275      *               <ul>
276      *
277      *               <li>file - the path is a fully-qualified filesystem
278      *               path.</li>
279      *
280      *               <li>servlet - the path is a servlet-context relative
281      *               path.</li>
282      *
283      *               <li>classpath - the path is a classpath-relative
284      *               path.</li>
285      *
286      *               </ul>
287      * @return a URL pointing to the given path in the given mechanism.
288      * @throws java.io.FileNotFoundException
289      * @throws java.net.MalformedURLException
290      */
291     protected URL getConfigURL(String path, String source)
292         throws IOException {
293         if (SOURCE_CLASSPATH.equals(source)) {
294             return this.getClassPathURL(path);
295         }
296 
297         if (SOURCE_FILE.equals(source)) {
298             return this.getFileURL(path);
299         }
300 
301         if (SOURCE_SERVLET.equals(source)) {
302             return this.getServletContextURL(path);
303         }
304 
305         // TODO Internationalize msg
306         throw new IllegalArgumentException("ConfigSource " + source
307             + " is not recognized");
308     }

310     /**
311      * Given a string, return a URL to a classpath resource of that name.
312      *
313      * @param path a Classpath-relative string identifying a resource.
314      * @return a URL identifying the resource on the classpath. TODO Do we
315      *         need to be smarter about ClassLoaders?
316      */
317     protected URL getClassPathURL(String path) {
318         return getClass().getClassLoader().getResource(path);
319     }
320 
321     /**
322      * Given a string, return a URL to a Servlet Context resource of that
323      * name.
324      *
325      * @param path a Classpath-relative string identifying a resource.
326      * @return a URL identifying the resource in the Servlet Context
327      * @throws MalformedURLException
328      */
329     protected URL getServletContextURL(String path)
330         throws IOException {
331         return this.servlet.getServletContext().getResource(path);
332     }
333 
334     /**
335      * Given a string, return a URL to a Filesystem resource of that name.
336      *
337      * @param path a path to a file.
338      * @return a URL identifying the resource in the in the file system.
339      * @throws MalformedURLException
340      * @throws FileNotFoundException
341      */
342     protected URL getFileURL(String path)
343         throws IOException {
344         File file = new File(path);
345 
346         return file.toURI().toURL();
347     }

389     /**
390      * This method is called after the Digester runs to store the generated
391      * object somewhere.  This implementation places the given object into the
392      * ServletContext under the attribute name as defined in
393      * <code>key</code>.
394      *
395      * @param obj The object to save.
396      */
397     protected void storeGeneratedObject(Object obj) {
398         log.debug("Put [{}] into application context [key:{}]",
399             obj, this.key);
400         this.servlet.getServletContext().setAttribute(this.getKey(), obj);
401     }
